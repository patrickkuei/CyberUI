## Quick Start

### Installation

```
npm install cyberui
```

### Basic Usage

```jsx
import React from 'react';
import { CircularProgress, Notification } from 'cyberui';

function App() {
  return (
    <div>
      <CircularProgress progress={75} size="large" />
      <Notification
        type="success"
        title="System Online"
        message="All systems operational"
      />
    </div>
  );
}
```
